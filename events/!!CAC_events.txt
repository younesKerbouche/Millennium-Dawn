add_namespace = cac_event

# Company is bankrupt
country_event = {
	id = cac_event.1
	title = cac_event.1.t
	desc = cac_event.1.d
	picture = GFX_foreclosure
	is_triggered_only = yes

	# Buyout Company
	option = {
		name = cac_event.1.a
		log = "[GetDateText]: [This.GetName]: cac_event.1.a executed"

		set_temp_variable = { cac_money_change = CAC_companies_money^i }

		set_temp_variable = { treasury_change = cac_money_change }
		modify_treasury_effect = yes

		multiply_temp_variable = { cac_money_change = -1 }

		set_temp_variable = { cac_slot = i }
		CAC_modify_money_effect = yes

		ai_chance = {
			base = 1
		}
	}

	option = {
		name = cac_event.1.b
		log = "[GetDateText]: [This.GetName]: cac_event.1.b executed"

		set_temp_variable = { var_to_change = -1 }

		set_temp_variable = { cac_factories_change = var_to_change }
		set_temp_variable = { cac_slot = i }
		set_temp_variable = { cac_ignore_value = 1 }
		CAC_modify_factories_effect = yes

		multiply_temp_variable = { var_to_change = -10 }

		set_temp_variable = { cac_money_change = var_to_change }
		set_temp_variable = { cac_slot = i }
		CAC_modify_money_effect = yes

		ai_chance = {
			base = 1
		}
	}

	# Do nothing
	option = {
		name = cac_event.1.c
		log = "[GetDateText]: [This.GetName]: cac_event.1.c executed"

		set_temp_variable = { cac_slot = i }
		CAC_remove_company_by_slot = yes

		ai_chance = {
			base = 1
		}
	}
}
