# Zaghawa Protests (Level 1)
country_event = {
	id = chad_tribal.1
	title = chad_tribal.1.t
	desc = chad_tribal.1.desc
	picture = GFX_african_demonstration

	trigger = {
		original_tag = CHA
		CHA_zaghawa_protest_level = yes
		NOT = { has_country_flag = CHA_zaghawa_protest_warning }
	}

	mean_time_to_happen = {
		days = 30
	}

	immediate = {
		set_country_flag = CHA_zaghawa_protest_warning
	}

	# Try to appease
	option = {
		name = chad_tribal.1.a
		add_political_power = -50
		add_to_variable = { CHA_zaghawa_loyalty = 10 }
		clamp_variable = { var = CHA_zaghawa_loyalty min = 0 max = 100 }
		CHA_update_tribal_resource_modifiers = yes
		ai_chance = { factor = 70 }
	}

	# Ignore protests
	option = {
		name = chad_tribal.1.b
		add_stability = -0.03
		add_to_variable = { CHA_zaghawa_loyalty = -5 }
		clamp_variable = { var = CHA_zaghawa_loyalty min = 0 max = 100 }
		CHA_update_tribal_resource_modifiers = yes
		ai_chance = { factor = 30 }
	}
}

# Zaghawa Partisans (Level 2)
country_event = {
	id = chad_tribal.2
	title = chad_tribal.2.t
	desc = chad_tribal.2.desc
	picture = GFX_african_rebels

	trigger = {
		original_tag = CHA
		CHA_zaghawa_partisan_level = yes
		NOT = { has_country_flag = CHA_zaghawa_partisan_warning }
	}

	mean_time_to_happen = {
		days = 45
	}

	immediate = {
		set_country_flag = CHA_zaghawa_partisan_warning
		add_ideas = CHA_tribal_partisans
	}

	# Military crackdown
	option = {
		name = chad_tribal.2.a
		add_political_power = -75
		add_war_support = 0.05
		add_to_variable = { CHA_zaghawa_loyalty = -10 }
		clamp_variable = { var = CHA_zaghawa_loyalty min = 0 max = 100 }
		CHA_update_tribal_resource_modifiers = yes
		ai_chance = { factor = 40 }
	}

	# Negotiate
	option = {
		name = chad_tribal.2.b
		add_political_power = -100
		add_to_variable = { CHA_zaghawa_loyalty = 15 }
		clamp_variable = { var = CHA_zaghawa_loyalty min = 0 max = 100 }
		remove_ideas = CHA_tribal_partisans
		CHA_update_tribal_resource_modifiers = yes
		ai_chance = { factor = 60 }
	}
}

	id = chad_tribal.3

		clamp_variable = { var = CHA_zaghawa_loyalty min = 0 max = 100 }
		ai_chance = { factor = 50 }
	}
}
