counter_terror_nation_startup = {
	set_variable = { internal_terror_threat = 0 }
	set_variable = { internal_terror_radicalization = 0 }
	set_variable = { external_terror_threat = 0 }
	set_variable = { external_terror_radicalization = 0}
}

counter_terror_core = {
	if = {
		limit = {
			has_variable = social_conservatism_government
		}
		social_conservatism_adjust = yes
	}
}
