remove_civilian_industry_effect = {
	#log = "ecovent100 eff: Lose ยงR[?reduction_factor|%0]ยง! of civilian factories."
	#log = "Ind: [?reduction_factor] [?reduction_factor|%0]"
	custom_effect_tooltip = remove_civilian_industry_effect_tt
	hidden_effect = {
		set_variable = { ind_factories = num_of_civilian_factories }
		log = "[This.GetName] Removing Industry, Factories: [?ind_factories], reduction factor: [?reduction_factor]"
		multiply_variable = {
			var = ind_factories
			value = reduction_factor
		}
		set_variable = { iterations = 0 }
		civ_factory_remove_loop_effect = yes
	}
}

civ_factory_remove_loop_effect = {
	hidden_effect = {
		log = "[This.GetName] Starting Removal Loop"
		if = {
			log = "[This.GetName] Removal Loop. Iteration: [?iterations]"
			limit = {
				check_variable = {
					var = iterations
					value = ind_factories
					compare = less_than
				}
			}
			random_owned_state = {
				remove_building = {
					type = industrial_complex
					level = 1
				}
				log = "[This.GetName] Removing industry in [This.GetName], [This.GetName]"
			}
			add_to_variable = {
				var = iterations
				value = 1
			}
			civ_factory_remove_loop_effect = yes
		}
	}
}

migrate_industry_to_random_effect = {
	hidden_effect = {
		random_country = {
			limit = {
				NOT = {
					has_idea = depression
					has_idea = recession
					has_idea = stagnation
				}
				has_war = no
				#num_of_controlled_states > ROOT #Should stop one-state country getting 50 factories. #Doesn't appear to work with scope reference
			}
			save_event_target_as = relocation_location
		}
	}
	custom_effect_tooltip = migrate_industry_to_random_effect_tt
	hidden_effect = {
		set_variable = { ind_factories = num_of_civilian_factories }
		log = "[This.GetName] Relocating Industry, Factories: [?ind_factories], reduction factor: [?reduction_factor]"
		multiply_variable = {
			var = ind_factories
			value = reduction_factor
		}
		set_variable = { iterations = 0 }
		civ_factory_remove_loop_effect = yes
		event_target:relocation_location = {
			country_event = {
				id = econvent.102
				days = 1
				random_hours = 24
			}
		}
	}
}

civ_factory_relocate_from_from_build_loop_effect = {
	hidden_effect = {
		log = "[This.GetName] Starting Construction Loop"
		if = {
			log = "[This.GetName] Construction Loop. Iteration: [?From.iterations]"
			limit = {
				FROM = { #To get ind_factories var
					check_variable = {
						var = iterations
						value = ind_factories
						compare = less_than
					}
				}
			}
			random_owned_state = {
				if = {
					limit = {
						free_building_slots = {
							building = industrial_complex
							size > 1
							include_locked = no
						}
					}
					add_building_construction = {
						type = industrial_complex
						level = 1
						instant_build = yes
					}
				}
				else = {
					add_extra_state_shared_building_slots = 1
					add_building_construction = {
						type = industrial_complex
						level = 1
						instant_build = yes
					}
				}
				log = "[This.GetName] Constructing industry in [This.GetName], [This.GetName]"
			}
			FROM = {
				add_to_variable = {
					var = iterations
					value = 1
				}
			}
			civ_factory_relocate_from_from_build_loop_effect = yes
		}
	}
}

# Set/Remove Improved Trade Agreement
# sender_nation --- the nation sending the trade agreement
# receiver_nation --- Nation retrieving the agreement
# remove_agreement -- OPTIONAL (Set to 1)
set_improved_trade_agreement = {
	if = {
		limit = {
			var:receiver_nation = {
				var:sender_nation = {
					NOT = { has_country_flag = trade_agreement@PREV }
				}
			}
			NOT = { check_variable = { remove_agreement = 1 } }
		}
		var:receiver_nation = {
			var:sender_nation = {
				add_opinion_modifier = { target = PREV modifier = mutual_trade_agreement }
				add_opinion_modifier = { target = PREV modifier = mutual_trade_opinion }
				reverse_add_opinion_modifier = { target = PREV modifier = mutual_trade_opinion }
				reverse_add_opinion_modifier = { target = PREV modifier = mutual_trade_agreement }

				hidden_effect = {
					add_to_variable = { signed_trade_agreements = 1 }
					set_country_flag = trade_agreement@PREV
					var:receiver_nation = {
						add_to_variable = { signed_trade_agreements = 1 }
						set_country_flag = trade_agreement@PREV
					}
				}
			}
		}
		custom_effect_tooltip = set_improved_trade_agreement_tt
	}
	if = {
		limit = {
			var:receiver_nation = {
				var:sender_nation = {
					has_country_flag = trade_agreement@PREV
				}
			}
			check_variable = { remove_agreement = 1 }
		}
		var:receiver_nation = {
			var:sender_nation = {
				remove_opinion_modifier = { target = PREV modifier = mutual_trade_agreement }
				remove_opinion_modifier = { target = PREV modifier = mutual_trade_opinion }
				add_opinion_modifier = { target = PREV modifier = broke_trade_agreement }

				clr_country_flag = trade_agreement@PREV
				add_to_variable = { signed_trade_agreements = -1 }
				var:receiver_nation = {
					remove_opinion_modifier = { target = PREV modifier = mutual_trade_agreement }
					remove_opinion_modifier = { target = PREV modifier = mutual_trade_opinion }
					add_opinion_modifier = { target = PREV modifier = broke_trade_agreement }
					clr_country_flag = trade_agreement@PREV
					add_to_variable = { signed_trade_agreements = -1 }
				}
			}
		}
		custom_effect_tooltip = remove_improved_trade_agreement_tt
	}
}

quarterly_inflation_calculation = {
	#Get and set the basic civilian economic growth for this quarter, and grab our old inflation rate
	set_temp_variable = { last_inflation_rate = inflation_rate_var }
	set_temp_variable = { gdpc_growth_inflation_adjuster = last_quarter_gdpc_growth_rate }

	#Grab the tax rate, and shift it to where it'll do the math right for us
	set_temp_variable = { tax_rate_modifier = tax_rate }
	multiply_temp_variable = { tax_rate_modifier = 0.01 }

	#Adjust the inflation rate based on GDP/C + Taxes
	set_temp_variable = { inflation_rate_adjuster = gdpc_growth_inflation_adjuster }
	log = "Pre Tax Rate Inflation Adjusted [?inflation_rate_adjuster]"
	multiply_temp_variable = { gdpc_growth_inflation_adjuster = tax_rate_modifier }
	subtract_from_temp_variable = { inflation_rate_adjuster = gdpc_growth_inflation_adjuster }
	log = "Post Tax Rate Inflation Adjusted [?inflation_rate_adjuster]"

	#Adjust inflation further by the base GDP/C Amount
	#Take the GDP/C and multiply it by the growth rate
	#This means that insanely high growth can springboard inflation, but slower / steadier growth raises it in a controlled manner
	set_temp_variable = { gdpc_inflation_adjuster = gdp_per_capita }
	multiply_temp_variable = { gdpc_inflation_adjuster = 0.01 }
	multiply_temp_variable = { gdpc_inflation_adjuster = last_quarter_gdpc_growth_rate }

	log = "Pre GDP/C Inflation Adjusted - [?inflation_rate_adjuster]"
	add_to_temp_variable = { inflation_rate_adjuster = gdpc_inflation_adjuster }
	log = "Post GDP/C Inflation Adjusted - [?inflation_rate_adjuster]"

	#Adjust inflation by economic cycle
	#Faster economic growth means more money and spending, which drives up the price of goods
	#This in turn pushes inflation. This should also scale on economic size
	if = {
		limit = { has_idea = depression }
		set_temp_variable = { cycle_inflation_adjustment = -0.035 }
	}else_if = {
		limit = { has_idea = recession }
		set_temp_variable = { cycle_inflation_adjustment = -0.025 }
	}else_if = {
		limit = { has_idea = stagnation }
		set_temp_variable = { cycle_inflation_adjustment = -0.015 }
	}else_if = {
		limit = { has_idea = stable_growth }
		set_temp_variable = { cycle_inflation_adjustment = 0 }
	}else_if = {
		limit = { has_idea = fast_growth }
		set_temp_variable = { cycle_inflation_adjustment = 0.015 }
	}else = {
		set_temp_variable = { cycle_inflation_adjustment = 0.025 }
	}

	set_temp_variable = { economic_expense_multiplier = display_expense }
	multiply_temp_variable = { economic_expense_multiplier = 0.01 }
	multiply_temp_variable = { cycle_inflation_adjustment = economic_expense_multiplier }

	log = "Pre Econ Cycle Inflation Adjustment [?inflation_rate_adjuster]"
	add_to_temp_variable = { inflation_rate_adjuster = cycle_inflation_adjustment }
	log = "Post Econ Cycle Inflation Adjustment [?inflation_rate_adjuster]"

	#Finally, adjust inflation based on the budget deficit. Larger deficit means more government effort in the economy
	#Typically, this would be by passively printing money/removing currency, which this system uses the deficit / surplus to simulate
	set_temp_variable = { treasury_inflation_adjust = treasury_rate }
	#Multiply here to get the proper addition state; negative budget increases inflation, positive budget decreases it
	#Then reduce it to the proper spot for the math
	multiply_temp_variable = { treasury_inflation_adjust = -1 }
	multiply_temp_variable = { treasury_inflation_adjust = 0.01 }
	#Reduce the GDP/C Modifier further here, this is just to help keep the math sane
	#Scale treasury inflation by GDP/C and growth rate (gdpc_inflation_adjuster was set earlier)
	set_temp_variable = { gdpc_treasury_scaling = gdpc_inflation_adjuster }
	multiply_temp_variable = { gdpc_treasury_scaling = 0.01 }
	multiply_temp_variable = { treasury_inflation_adjust = gdpc_treasury_scaling }

	add_to_temp_variable = { inflation_rate_adjuster = treasury_inflation_adjust }

	#Apply External Modifiers Here
	#If you've printed money, increase inflation, scaled by how many times you printed it + how much money it was
	#This is not perfect math, but it simply increases whatever economic state you are in, so it can make inflation or deflation worse
	set_temp_variable = { print_money_base_multiplier = 1 }
	add_to_temp_variable = { print_money_base_multiplier = print_money_base_add }
	set_variable = { print_money_base_add = 0 }

	multiply_temp_variable = { inflation_rate_adjuster = print_money_base_multiplier }

	#Calcuation of the quarterly rate finished, now move to calculate the yearly average off those numbers
	#Check to make sure the array has only 4 items, and add up the value
	set_temp_variable = { total_elements = year_quarter_inflation_tracker^num }

	if = { limit = { check_variable = { total_elements > 3 } }
		remove_from_array = {
			array = year_quarter_inflation_tracker
			index = 0
		}
	}

	#Add this to the array, so we can average yearly inflation. this should stop rapid swings up and down, slowing movement
	add_to_array = { year_quarter_inflation_tracker = inflation_rate_adjuster }
	set_temp_variable = { new_inflation_rate = 0 }
	#Reset the tracker here; this will prevent divide by 0, and give the most current number before division
	set_temp_variable = { total_elements = year_quarter_inflation_tracker^num }

	for_each_loop = {
		array = year_quarter_inflation_tracker
		value = v

		add_to_temp_variable = { new_inflation_rate = v }
	}

	#Divide to get the average number
	divide_temp_variable = { new_inflation_rate = total_elements }

	#Maximum inflation movement per quarter of 10%. This means that the max you can move in a year is 40% in either direction
	#This is a balance mechanic, both for gameplay, and for when you eat a whole country and your finances go crazy. On quick quarterly spike, and then back down
	set_temp_variable = { inflation_rate_max = last_inflation_rate }
	add_to_temp_variable = { inflation_rate_max = 0.1 }
	set_temp_variable = { inflation_rate_min = last_inflation_rate }
	add_to_temp_variable = { inflation_rate_min = -0.1 }

	clamp_temp_variable = {
		var = new_inflation_rate
		min = inflation_rate_min
		max = inflation_rate_max
	}

	set_variable = { inflation_rate_var = new_inflation_rate }
	if = { limit = { is_debug = yes }
		log = "[GetDateText]: [THIS.GetName]: Inflation Update: Current Inflation Rate: [?THIS.inflation_rate_var|%] - Last Four Quarters: [?THIS.year_quarter_inflation_tracker^0|%] - [?THIS.year_quarter_inflation_tracker^1|%] - [?THIS.year_quarter_inflation_tracker^2|%] - [?THIS.year_quarter_inflation_tracker^3|%]"
	}
	calculate_inflation_effects = yes
}
