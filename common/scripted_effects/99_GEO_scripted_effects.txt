GEO_update_energy_problems_idea = {
	if = {
		limit = {
			has_idea = GEO_gaz_energy_problems_idea
		}
		swap_ideas = {
			remove_idea = GEO_gaz_energy_problems_idea
			add_idea = GEO_energy_problems
		}
	}
	else_if = {
		limit = {
			has_idea = GEO_gaz_problems
		}
		remove_ideas = GEO_gaz_problems
	}
}
