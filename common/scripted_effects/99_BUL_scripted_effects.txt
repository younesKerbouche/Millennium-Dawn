			}
			set_popularities = {
				democratic = 5

				fascism = 0
				communism = 5
				nationalist = 90
			}
			inherit_technology = FYR
			clear_array = ruling_party
			clear_array = gov_coalition_array
			add_to_array = {
				ruling_party = 23
			}
			set_temp_variable = {
				current_cons_popularity = party_pop_array^23
			}
			subtract_from_variable = {
				party_pop_array^23 = current_cons_popularity
			}
			recalculate_party = yes
