ideas = {
	country = {
		# For developed nations (Western Europe, North America, Australia/NZ, Japan, South Korea)
		# Applies slight penalty to reflect diminishing returns at productivity frontier
		# Removes when global average rises OR individual nation significantly outperforms
		developed_economy_transition = {
			name = developed_economy_transition
			allowed_civil_war = { always = yes }
			on_add = { log = "[GetDateText]: [Root.GetName]: add idea developed_economy_transition" }
			on_remove = { log = "[GetDateText]: [Root.GetName]: removed developed_economy_transition (global avg: [?global.average_world_productivity], own productivity: [?overall_productivity])" }

			picture = generic_economy

			modifier = {
				production_speed_buildings_factor = -0.15
				productivity_growth_modifier = -0.10
			}

			# Removes when global average rises OR nation significantly outperforms
			cancel = {
				OR = {
					check_variable = { global.average_world_productivity > 850 }
					check_variable = { overall_productivity > 1200 } # Individual escape: 20% above Europe start (1000)
				}
			}
		}

		# For emerging economies (BRICS, Mexico, Turkey, Eastern Europe, etc.)
		# Provides growth bonus to help catch up to developed nations
		emerging_economy_transition = {
			name = emerging_economy_transition
			allowed_civil_war = { always = yes }
			on_add = { log = "[GetDateText]: [Root.GetName]: add idea emerging_economy_transition" }
			on_remove = { log = "[GetDateText]: [Root.GetName]: removed emerging_economy_transition (GDP/C: [?gdp_per_capita], productivity: [?overall_productivity])" }

			picture = generic_economy

			modifier = {
				productivity_growth_modifier = 0.15
				production_speed_buildings_factor = 0.05
			}

			# Removes when country catches up to global average or becomes developed
			cancel = {
				OR = {
					check_variable = { overall_productivity > global.average_world_productivity }
					check_variable = { gdp_per_capita > 24.999 } # gdp >= 25, matches assignment
				}
			}
		}

		# For least developed nations (Sub-Saharan Africa, etc.)
		# Provides strong catch-up bonus to help developing nations grow
		developing_economy_catch_up = {
			name = developing_economy_catch_up
			allowed_civil_war = { always = yes }
			on_add = { log = "[GetDateText]: [Root.GetName]: add idea developing_economy_catch_up" }
			on_remove = { log = "[GetDateText]: [Root.GetName]: removed developing_economy_catch_up (GDP/C: [?gdp_per_capita], productivity: [?overall_productivity])" }

			picture = generic_economy

			modifier = {
				productivity_growth_modifier = 0.25
				production_speed_buildings_factor = 0.10
			}

			# Removes when country reaches emerging economy status
			cancel = {
				OR = {
					check_variable = { gdp_per_capita > 15 }
					check_variable = { overall_productivity > 700 } # Raised from 600 to avoid premature triggers
				}
			}
		}
	}
}
