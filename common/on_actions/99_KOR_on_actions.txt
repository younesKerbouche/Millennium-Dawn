on_actions = {
	on_weekly_KOR = {
		random_events = {
			100 = 0

			1000 = korea.16
			1000 = korea.17
			1000 = korea.22
		}
	}

	on_weekly_NKO = {
		random_events = {
			100 = 0

			1000 = korea.22

		}
	}
	on_annex = {
		effect = {
			if = {
				limit = {
					OR = {
						AND = {
							ROOT = NKO
							FROM = KOR
						}
						AND = {
							ROOT = KOR
							FROM = NKO
						}
					}
				}
				if = {
					limit = { KOR = { exists = yes } }
					KOR = {
						set_country_flag = korea_peninsula_reunited
						add_ideas = KOR_reunification_shocks
						add_ideas = KOR_divided_military
						add_ideas = KOR_mass_migration
						add_ideas = KOR_fragmented_governance
						add_ideas = KOR_uneven_economy1
					}
				}
				else_if = {
					limit = { NKO = { exists = yes } }
					NKO = {
						set_country_flag = korea_peninsula_reunited
						add_ideas = KOR_reunification_shocks
						add_ideas = KOR_divided_military
						add_ideas = KOR_mass_migration
						add_ideas = KOR_fragmented_governance
						add_ideas = KOR_uneven_economy1
					}
				}
			}
		}
	}
}
