faction_goal_incorporate_syria = {
	name = incorporate_syria_name
	description = incorporate_syria_desc
	category = short_term
	group = FOCUS_FILTER_POLITICAL_CHARACTER
	visible = {
		has_faction_template = faction_template_csto
		country_exists = SYR
		SYR = {
			NOT = {
				is_in_faction = yes
			}
		}
	}
	completed = {
		SYR = {
			is_in_faction_with = ROOT
		}
	}
	complete_effect = {
		add_faction_power_projection = 200
		custom_effect_tooltip = generic_skip_one_line_tt
		custom_effect_tooltip = THE_FACTION_LEADER
		add_faction_initiative = 1
		every_faction_member = {
			limit = {
				NOT = {
					original_tag = SYR
				}
			}
			set_temp_variable = { percent_change = 2 }
			set_temp_variable = { tag_index = THIS }
			set_temp_variable = { influence_target = SYR }
			change_influence_percentage = yes
		}
	}
	ai_will_do = {
		base = 1
	}
}
