	libya_nationalise_oil_fields = {

		icon = eng_trade_unions_demand

		cost = 50

		days_remove = 30
		days_re_enable = 45

		visible = {
			check_variable = { lba_oil_nationalisation < 1 }
			NOT = { has_country_flag = libya_noc_dismantled }
		}

		available = {
			NOT = { has_decision = libya_sell_oil_fields }
		}

		complete_effect = {
			log = "[GetDateText]: [Root.GetName]: Decision complete libya_nationalise_oil_fields"
			set_temp_variable = { treasury_change = gdp_total }
			multiply_temp_variable = { treasury_change = -0.1 }
			modify_treasury_effect = yes
		}

		remove_effect = {
			log = "[GetDateText]: [Root.GetName]: Decision remove libya_nationalise_oil_fields"
			set_temp_variable = { percent_change = 3 }
			change_domestic_influence_percentage = yes
			libya_increase_nationalisation_5 = yes
		}

		ai_will_do = {
			factor = 1
			#If we have loads of money, nationalise
			modifier = {
				factor = 5
				check_variable = { interest_rate < 2.5 }
			}
			#Liberals, libertarians and oligarchs only want to sell
			modifier = {
				factor = 0
				OR = {
					is_in_array = { ruling_party = 2 }
					is_in_array = { ruling_party = 15 }
					is_in_array = { ruling_party = 16 }
				}
			}
			#Communists always want to nationalise
			modifier = {
				factor = 0
				OR = {
					is_in_array = { ruling_party = 4 }
					is_in_array = { ruling_party = 19 }
					is_in_array = { ruling_party = 21 }
				}
				check_variable = { interest_rate < 8 }
			}
			#Buy if our influence is too low
			modifier = {
				factor = 5
				check_variable = { domestic_influence_amount < 30 }
			}
		}
	}

	libya_sell_oil_fields = {

		icon = gre_paying_ifc_debt

		cost = 50

		days_remove = 30
		days_re_enable = 45

		visible = {
			check_variable = { lba_oil_nationalisation > 0 }
			NOT = { has_country_flag = libya_noc_dismantled }
		}

		available = {
			NOT = { has_decision = libya_nationalise_oil_fields }
		}

		complete_effect = {
			log = "[GetDateText]: [Root.GetName]: Decision complete libya_sell_oil_fields"
			set_temp_variable = { treasury_change = gdp_total }
			multiply_temp_variable = { treasury_change = 0.075 }
			modify_treasury_effect = yes
		}

		remove_effect = {
			log = "[GetDateText]: [Root.GetName]: Decision remove libya_sell_oil_fields"
			random_country = {
				limit = {
					NOT = { original_tag = ROOT }
					any_owned_state = {
						oil > 0
					}
					NOT = { has_idea = non_power }
				}
				set_temp_variable = { percent_change = 2 }
				set_temp_variable = { tag_index = THIS }
				set_temp_variable = { influence_target = ROOT }
				change_influence_percentage = yes
			}
			libya_decrease_nationalisation_5 = yes
		}

		ai_will_do = {
			factor = 1
			#If we are broke, sell
			modifier = {
				factor = 5
				check_variable = { interest_rate > 9 }
			}
			#Liberals, libertarians and oligarchs only want to sell
			modifier = {
				factor = 5
				OR = {
					is_in_array = { ruling_party = 2 }
					is_in_array = { ruling_party = 15 }
					is_in_array = { ruling_party = 16 }
				}
			}
			#Communists don't sell
			modifier = {
				factor = 0
				OR = {
					is_in_array = { ruling_party = 4 }
					is_in_array = { ruling_party = 19 }
					is_in_array = { ruling_party = 21 }
				}
			}
			#Don't sell if our influence is too low
			modifier = {
				factor = 0
				check_variable = { domestic_influence_amount < 30 }
			}
		}
	}
