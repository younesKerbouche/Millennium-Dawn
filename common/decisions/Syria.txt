	syria_decision_fund_voluntary_organizations = {
		icon = GFX_decision_syr_money_button
		#available = {}
		visible = {
			has_completed_focus = SYR_address_ethnic_divides
		}
		days_remove = 120
		fixed_random_seed = no
		days_re_enable = 35
		complete_effect = {
			log = "[GetDateText]: [Root.GetName]: Decision syria_decision_fund_voluntary_organizations"
			add_timed_idea = {
				idea = syria_funding_voluntary_organizations
				days = 120
			}
		}
		remove_effect = {
			set_temp_variable = { voluntary_groups_chance = 0 }
			set_temp_variable = { syrian_centralization_minority_average = 0 }
			set_temp_variable = { syrian_number_of_minorities = 0 }
			if = {
				limit = { has_dynamic_modifier = { modifier = alawite_political_inclusiveness_dynamic_modifier scope = SYR } }
				add_to_temp_variable = { syrian_centralization_minority_average = alawite_status }
				add_to_temp_variable = { syrian_number_of_minorities = 1 }
			}
			if = {
				limit = { has_dynamic_modifier = { modifier = druze_political_inclusiveness_dynamic_modifier scope = SYR } }
				add_to_temp_variable = { syrian_centralization_minority_average = druze_status }
				add_to_temp_variable = { syrian_number_of_minorities = 1 }
			}
			if = {
				limit = { has_dynamic_modifier = { modifier = kurdish_political_inclusiveness_dynamic_modifier scope = SYR } }
				add_to_temp_variable = { syrian_centralization_minority_average = kurdish_status }
				add_to_temp_variable = { syrian_number_of_minorities = 1 }
			}
			if = {
				limit = { check_variable = { syrian_number_of_minorities = 0 } }
				add_to_temp_variable = { syrian_centralization_minority_average = 80 }
			}
			else = {
				divide_temp_variable = { syrian_centralization_minority_average = syrian_number_of_minorities }
				divide_temp_variable = { syrian_centralization_minority_average = 1.25 }
			}
			add_to_temp_variable = { voluntary_groups_chance = syrian_centralization_minority_average }

			if = {
				limit = { is_in_array = { gov_coalition_array = 17 } }
				add_to_temp_variable = { voluntary_groups_chance = 20 }
			}
			set_temp_variable = { remaining_chance = 100 }
			subtract_from_temp_variable = { remaining_chance = voluntary_groups_chance }
			clamp_temp_variable = {
				var = remaining_chance
				min = 0
				max = 100
			}
			random_list = {
				1 = {
					modifier = {
						factor = remaining_chance
					}
					SYR = { country_event = { id = SyriaFocus.92 days = 2 } }
				}
				1 = {
					modifier = {
						factor = voluntary_groups_chance
					}
					SYR = { country_event = { id = SyriaFocus.86 days = 2 } }
				}
			}
			custom_effect_tooltip = SYR_FUND_VOLUNTARY_ORGANIZATIONS_SUCCESS_EFFECTS
			custom_effect_tooltip = TT_SYR_BOP_SUCCESSFUL_FEDERALISM_EFFECT
		}
	}
