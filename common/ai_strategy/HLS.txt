HLS_italian_relations = {
	allowed = { original_tag = ITA }
	enable = {
		country_exists = HLS
		has_idea = christian
		HLS = { NOT = { has_government = fascism } }
	}
	abort_when_not_enabled = yes

	ai_strategy = { type = befriend id = "HLS" value = 100 }
	ai_strategy = { type = alliance id = "HLS" value = 100 }
	ai_strategy = { type = support id = "HLS" value = 100 }
}
