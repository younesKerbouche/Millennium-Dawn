defined_text = {
    name = ALG_power_struggle_state

    # Resistance dominating
    text = {
        trigger = { check_variable = { ALG_people_grievance > 65 } }
        localization_key = ALG_power_struggle_resistance_dominant
    }

    # Army dominating
    text = {
        trigger = { check_variable = { ALG_army_influence > 65 } }
        localization_key = ALG_power_struggle_army_dominant
    }

    # Close contest (difference > 15 and both at decent level)
    text = {
        trigger = {
            check_variable = { ALG_people_grievance > 30 }
            check_variable = { ALG_army_influence > 30 }
            # difference check: people stronger by 15
            check_variable = { ALG_people_grievance > ALG_army_influence value = 15 }
        }
        localization_key = ALG_power_struggle_resistance_rising
    }

    text = {
        trigger = {
            check_variable = { ALG_people_grievance > 30 }
            check_variable = { ALG_army_influence > 30 }
            # army stronger by 15
            check_variable = { ALG_army_influence > ALG_people_grievance value = 15 }
        }
        localization_key = ALG_power_struggle_army_rising
    }

    # Default / low-activity fallback
    text = {
        trigger = { always = yes }
        localization_key = ALG_power_struggle_stable
    }
}
