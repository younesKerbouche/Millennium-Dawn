scripted_gui = {
	construction_production_speed_menu = {
		context_type = player_context
		window_name = "construction_building_production_speed_container"
		parent_window_token = construction_tab
		dirty = global.update_construction_speed_ui

		effects = {
			toggle_construction_building_speed_click = {
				if = {
					limit = { NOT = { has_country_flag = construction_speeds_visible_in_menu } }
					set_country_flag = construction_speeds_visible_in_menu
				}
				else = {
					clr_country_flag = construction_speeds_visible_in_menu
				}
				add_to_variable = { global.update_construction_speed_ui = 1 }
			}
		}

		properties = {
			toggle_construction_building_speed = { image = "[building_construction_speed_toggle]" }
		}

		triggers = {
			building_construction_speeds_visible = { has_country_flag = construction_speeds_visible_in_menu }
		}
		ai_enabled = { always = no }
	}
	construction_speed_view_menu = {
		context_type = player_context
		window_name = "building_construction_speeds"
		parent_window_token = construction_tab
		dirty = global.update_construction_speed_ui
		visible = {
			has_country_flag = construction_speeds_visible_in_menu
		}

		ai_enabled = { always = no }
	}
}
