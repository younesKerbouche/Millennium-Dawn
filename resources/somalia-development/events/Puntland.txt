	fire_only_once = yes

	option = {
		name = puntland.2.a
		log = "[GetDateText]: [This.GetName]: puntland.2.a executed"

		add_stability = 0.05

		ai_chance = {
			factor = 1
			modifier = {
				has_game_rule = {
					rule = PUN_ai_behavior

			}

	}

	option = {
		name = puntland.2.b

		add_ideas = PUN_pirate_navy
		set_country_flag = PUN_is_pirate
		add_to_variable = { var = own_crewmembers value = 1 }

		custom_effect_tooltip = PUN_increase_enemy_ship_1_tt
		add_to_variable = { var = pirates_morale value = 15 }
		custom_effect_tooltip = PUN_morale_increases_15_tt

			limit = {
				NOT = { has_tech = corvette_1 }
			}

				corvette_1 = 1
			}
		}

		ai_chance = {
			factor = 0
			modifier = {
				has_game_rule = {
					rule = PUN_AI_BEHAVIOR

				}
				add = 20

		}
	}
